 13class NearestNeighbors(abc.ABC):
 14    """
 15    Abstract Base Class for Nearest Neighbors implementations tailored for ESA.
 16
 17    This interface defines the contract for hybrid neighbor search engines that manage
 18    two sets of points:
 19    1. **Static Set:** Fixed points (anchors/obstacles) that accumulate over time.
 20    2. **Active Set:** A transient batch of points currently being optimized.
 21
 22    The implementations must support efficient queries of Active points against the
 23    union of both sets: $P_{query} \\in Active \\to \\{P_{static} \\cup P_{active}\\}$.
 24    """
 25
 26    def __init__(self, dimension: int):
 27        self.dimension = dimension
 28
 29    @abc.abstractmethod
 30    def add_static(self, points: np.ndarray) -> None:
 31        """
 32        Adds points to the persistent static set.
 33
 34        These points represent obstacles or previously consolidated solutions that the active
 35        batch must avoid.
 36
 37        Args:
 38            points (np.ndarray): Array of shape $(N, D)$ to add to the index.
 39        """
 40        pass
 41
 42    @abc.abstractmethod
 43    def set_active(self, points: np.ndarray) -> None:
 44        """
 45        Sets the current batch of active points for optimization.
 46
 47        These points are transient and updated iteratively. Setting a new active batch
 48        replaces the previous one.
 49
 50        Args:
 51            points (np.ndarray): Array of shape $(M, D)$ representing the moving particles.
 52        """
 53        pass
 54
 55    @abc.abstractmethod
 56    def consolidate(self) -> None:
 57        """
 58        Merges the current active points into the static set.
 59
 60        This is typically called when the optimization of the current batch converges.
 61        The active points become permanent obstacles for future batches.
 62
 63        $ S_{new} = S_{old} \\cup A_{current} $
 64        """
 65        pass
 66
 67    @abc.abstractmethod
 68    def query_nn(self, k: int) -> tuple[np.ndarray, np.ndarray]:
 69        """
 70        Finds the $k$-nearest neighbors for the ACTIVE batch against the full set (Static + Active).
 71
 72        For each point $x_i$ in the active set, this computes:
 73        $ N(x_i) = \\{ y \\in (S \\cup A) \\mid \\text{rank}(||x_i - y||) \\le k \\} $
 74
 75        The results serve as the basis for k-NN repulsive forces.
 76
 77        Args:
 78            k (int): The number of neighbors to retrieve.
 79
 80        Returns:
 81            tuple[np.ndarray, np.ndarray]:
 82                - **Indices**: Shape $(M, k)$, global indices of neighbors.
 83                - **Distances**: Shape $(M, k)$, Euclidean distances to neighbors.
 84        """
 85        pass
 86
 87    @abc.abstractmethod
 88    def query_static(
 89        self, query_points: np.ndarray, k: int
 90    ) -> tuple[np.ndarray, np.ndarray]:
 91        """
 92        Finds the $k$-nearest neighbors for arbitrary query points against the STATIC set only.
 93
 94        Used mainly for initialization heuristics where we only care about distance to
 95        established obstacles, ignoring other nascent points.
 96
 97        Args:
 98            query_points (np.ndarray): The query coordinates $(Q, D)$.
 99            k (int): Number of neighbors.
100
101        Returns:
102            tuple[np.ndarray, np.ndarray]: Indices and distances.
103        """
104        pass
105
106    @abc.abstractmethod
107    def range_search(self, radius: float) -> tuple[np.ndarray, np.ndarray]:
108        """
109        Finds all neighbors within a fixed `radius` for the ACTIVE batch.
110
111        This returns a dense matrix representation suitable for vectorized force calculation.
112        For active point $x_i$ and potential neighbor $y_j$:
113
114        $ M_{ij} = \\begin{cases} ||x_i - y_j|| & \\text{if } ||x_i - y_j|| < R \\\\ 0 & \\text{otherwise} \\end{cases} $
115
116        Args:
117            radius (float): The cutoff distance $R$.
118
119        Returns:
120            tuple[np.ndarray, np.ndarray]:
121                - **Distances**: Dense matrix $(M, N_{total})$. Zero where no neighbor exists.
122                - **Mask**: Boolean matrix $(M, N_{total})$. True where $d < R$.
123        """
124        pass

126    @abc.abstractmethod
127    def clear(self) -> None:
128        """
129        Resets the internal state, clearing both static and active sets.
130        """
131        pass

134    @abc.abstractmethod
135    def total_count(self) -> int:
136        """
137        Returns the total number of points currently managed (Static + Active).
138        """

142class NumpyNN(NearestNeighbors):
143    """
144    Pure NumPy implementation using vectorized broadcasting and caching.
145
146    This class is optimized for small to medium datasets ($N < 5000$) where the overhead
147    of building tree structures or HNSW graphs outweighs the $O(N^2)$ cost of brute-force
148    matrix operations.
149
150    **Optimization Mechanism:**
151    It utilizes the squared Euclidean distance expansion to leverage BLAS Level 3 optimizations
152    (matrix multiplication):
153    $ ||A - B||^2 = ||A||^2 + ||B||^2 - 2A \\cdot B^T $
154
155    The term $||B||^2$ (static points squared norms) is cached and incrementally updated
156    to avoid recomputation during iterative optimization steps.
157    """
158
159    def __init__(self, dimension: int):
160        super().__init__(dimension)
161        self._static: np.ndarray = np.empty((0, dimension), dtype=np.float32)
162        self._static_sq: np.ndarray = np.empty((0, 1), dtype=np.float32)
163        self._active: np.ndarray = np.empty((0, dimension), dtype=np.float32)
164
165    def add_static(self, points: np.ndarray) -> None:
166        """
167        Adds points to the static set and updates the squared-norm cache.
168
169        Args:
170            points (np.ndarray): Points to add $(N, D)$.
171        """
172        if points.shape[1] != self.dimension:
173            raise ValueError(f"Dim mismatch: {points.shape[1]} vs {self.dimension}")
174
175        points = points.astype(np.float32)
176        self._static = np.vstack((self._static, points))
177
178        # Update cache: ||B||^2
179        points_sq = np.sum(points**2, axis=1, keepdims=True)
180        self._static_sq = np.vstack((self._static_sq, points_sq))
181
182    def set_active(self, points: np.ndarray) -> None:
183        """
184        Sets the active batch buffer.
185
186        Args:
187            points (np.ndarray): Active points $(M, D)$.
188        """
189        if points.shape[1] != self.dimension:
190            raise ValueError(f"Dim mismatch: {points.shape[1]} vs {self.dimension}")
191        self._active = points.astype(np.float32)
192
193    def consolidate(self) -> None:
194        """
195        Transfers active points to the static array and updates the norm cache.
196        """
197        if self._active.shape[0] > 0:
198            self.add_static(self._active)
199            self._active = np.empty((0, self.dimension), dtype=np.float32)
200
201    def clear(self) -> None:
202        """
203        Reallocates empty buffers for static and active sets.
204        """
205        self._static = np.empty((0, self.dimension), dtype=np.float32)
206        self._static_sq = np.empty((0, 1), dtype=np.float32)
207        self._active = np.empty((0, self.dimension), dtype=np.float32)
208
209    @property
210    def total_count(self) -> int:
211        """
212        The sum of rows in the static and active buffers.
213        """
214        return self._static.shape[0] + self._active.shape[0]
215
216    def _sq_dist_matrix_cached(
217        self, A: np.ndarray, B: np.ndarray, B_sq: np.ndarray
218    ) -> np.ndarray:
219        """Computes ||A - B||^2 using cached ||B||^2."""
220        A_sq = np.sum(A**2, axis=1, keepdims=True)
221        # A_sq + B_sq.T - 2AB
222        return A_sq + B_sq.T - 2 * np.dot(A, B.T)
223
224    def _sq_dist_matrix_uncached(self, A: np.ndarray, B: np.ndarray) -> np.ndarray:
225        """Standard ||A - B||^2 computation."""
226        A_sq = np.sum(A**2, axis=1, keepdims=True)
227        B_sq = np.sum(B**2, axis=1, keepdims=True)
228        return A_sq + B_sq.T - 2 * np.dot(A, B.T)
229
230    def _compute_full_sq_dists(self) -> np.ndarray:
231        """Computes merged squared distances from Active to (Static + Active)."""
232        n_active = self._active.shape[0]
233        n_static = self._static.shape[0]
234
235        # 1. Active vs Static (Using Cache)
236        if n_static > 0:
237            dists_s_sq = self._sq_dist_matrix_cached(
238                self._active, self._static, self._static_sq
239            )
240        else:
241            dists_s_sq = np.empty((n_active, 0), dtype=np.float32)
242
243        # 2. Active vs Active (Uncached)
244        dists_a_sq = self._sq_dist_matrix_uncached(self._active, self._active)
245
246        # 3. Merge
247        full_dists_sq = np.hstack((dists_s_sq, dists_a_sq))
248        return np.maximum(full_dists_sq, 0.0)
249
250    def query_nn(self, k: int) -> tuple[np.ndarray, np.ndarray]:
251        """
252        Performs exact k-NN search using dense matrix operations.
253
254        **Algorithm:**
255        1. Compute full distance matrix $D^2$ using cached norms ($M \\times N$).
256        2. Use `np.argpartition` (Introselect) to find the top-$k$ elements in $O(N)$ time.
257        3. Perform a small sort on the $k$ elements to return them in order.
258
259        Args:
260            k (int): Number of neighbors.
261
262        Returns:
263            tuple[np.ndarray, np.ndarray]: Indices and distances.
264        """
265        full_dists_sq = self._compute_full_sq_dists()
266        n_samples = full_dists_sq.shape[0]
267        k = min(k, full_dists_sq.shape[1])
268
269        # 1. Argpartition to find top-k (unsorted)
270        # Time: O(N * Total_Count)
271        unsorted_top_k_idx = np.argpartition(full_dists_sq, k - 1, axis=1)[:, :k]
272
273        # 2. Extract values (Optimized indexing)
274        # Create row indices: [[0,0,..], [1,1,..], ...]
275        row_indices = np.arange(n_samples)[:, None]
276
277        # Fancy indexing is often faster than take_along_axis for simple 2D arrays
278        top_dists_sq = full_dists_sq[row_indices, unsorted_top_k_idx]
279
280        # 3. Sort the top-k (Small sort: k * log k)
281        # We only sort the small k window, not the full array
282        local_sort_order = np.argsort(top_dists_sq, axis=1)
283
284        # 4. Final Gather
285        final_indices = unsorted_top_k_idx[row_indices, local_sort_order]
286        final_dists_sq = top_dists_sq[row_indices, local_sort_order]
287
288        return final_indices, np.sqrt(final_dists_sq)
289
290    def query_static(
291        self, query_points: np.ndarray, k: int
292    ) -> tuple[np.ndarray, np.ndarray]:
293        """
294        Queries the static set using cached squared norms.
295
296        Args:
297            query_points (np.ndarray): Query inputs.
298            k (int): Neighbors count.
299
300        Returns:
301            tuple[np.ndarray, np.ndarray]: Indices and distances.
302        """
303        if self._static.shape[0] == 0:
304            return (
305                np.zeros((len(query_points), k), dtype=int),
306                np.full((len(query_points), k), np.inf),
307            )
308
309        k = min(k, self._static.shape[0])
310        # Use cached static norms
311        dist_sq = self._sq_dist_matrix_cached(
312            query_points, self._static, self._static_sq
313        )
314        dist_sq = np.maximum(dist_sq, 0.0)
315
316        if k == 1:
317            final_idx = np.argmin(dist_sq, axis=1)[:, None]
318            final_dists = np.sqrt(np.take_along_axis(dist_sq, final_idx, axis=1))
319            return final_idx, final_dists
320
321        part_idx = np.argpartition(dist_sq, k - 1, axis=1)[:, :k]
322        top_dists_sq = np.take_along_axis(dist_sq, part_idx, axis=1)
323
324        sort_idx = np.argsort(top_dists_sq, axis=1)
325        final_dists = np.sqrt(np.take_along_axis(top_dists_sq, sort_idx, axis=1))
326        final_idx = np.take_along_axis(part_idx, sort_idx, axis=1)
327
328        return final_idx, final_dists
329
330    def range_search(self, radius: float) -> tuple[np.ndarray, np.ndarray]:
331        """
332        Computes the full distance matrix and applies a boolean mask.
333
334        Because `NumpyNN` is designed for dense regimes, this returns the complete
335        $M \\times N$ distance matrix and a boolean mask indicating valid neighbors ($d < R$).
336        This avoids the overhead of sparse formatting when connectivity is high.
337
338        Args:
339            radius (float): Cutoff radius.
340
341        Returns:
342            tuple[np.ndarray, np.ndarray]: (Distances, Mask).
343        """
344        # 1. Compute Full Matrix
345        full_dists_sq = self._compute_full_sq_dists()
346
347        # 2. Compute Mask
348        radius_sq = radius * radius
349        mask = full_dists_sq < radius_sq
350
351        # 3. Return (Dists, Mask) - No 'nonzero' or sorting required
352        # Return sqrt distances for metric compatibility
353        return np.sqrt(full_dists_sq), mask

356class FaissBaseNN(NearestNeighbors):
357    """
358    Base wrapper for Faiss index implementations.
359
360    This class handles the hybrid logic of querying a specialized Faiss index (for the
361    large Static set) and combining it with brute-force calculations for the (small)
362    Active set.
363
364    **Hybrid Query Logic:**
365    Since Faiss indices are often immutable or expensive to rebuild frequently, the
366    Active set is kept separate.
367    $ N(x) = \\text{SelectTopK}( \\text{FaissQuery}(x, S) \\cup \\text{BruteForce}(x, A) ) $
368    """
369
370    def __init__(self, index_static: faiss.Index, dimension: int):
371        super().__init__(dimension)
372        self._active: np.ndarray = np.empty((0, dimension), dtype=np.float32)
373        self._static_count = 0
374        self._index_static: typing.Any = index_static
375
376    def add_static(self, points: np.ndarray) -> None:
377        """
378        Adds vectors to the underlying Faiss index.
379
380        Args:
381            points (np.ndarray): Data to index.
382        """
383        data = np.ascontiguousarray(points.astype(np.float32))
384        if data.shape[1] != self.dimension:
385            raise ValueError(f"Dim mismatch: {data.shape[1]} vs {self.dimension}")
386        self._index_static.add(data)
387        self._static_count += data.shape[0]
388
389    def set_active(self, points: np.ndarray) -> None:
390        """
391        Updates the transient active batch.
392        """
393        if points.shape[1] != self.dimension:
394            raise ValueError("Dim mismatch")
395        self._active = np.ascontiguousarray(points.astype(np.float32))
396
397    def consolidate(self) -> None:
398        """
399        Adds the current active batch to the Faiss index and clears the active buffer.
400        """
401        if self._active.shape[0] > 0:
402            self.add_static(self._active)
403            self._active = np.empty((0, self.dimension), dtype=np.float32)
404
405    def clear(self) -> None:
406        """
407        Resets the Faiss index and local counters.
408        """
409        self._index_static.reset()
410        self._static_count = 0
411        self._active = np.empty((0, self.dimension), dtype=np.float32)
412
413    @property
414    def total_count(self) -> int:
415        """
416        Total points in Faiss index + active buffer size.
417        """
418        return self._static_count + self._active.shape[0]
419
420    def _merge_active_active_knn(
421        self, dists_s, idxs_s, k: int
422    ) -> tuple[np.ndarray, np.ndarray]:
423        n_active = self._active.shape[0]
424
425        # Active-Active (Brute Force)
426        A_sq = np.sum(self._active**2, axis=1, keepdims=True)
427        dists_a_sq = A_sq + A_sq.T - 2 * np.dot(self._active, self._active.T)
428        dists_a_sq = np.maximum(dists_a_sq, 0.0)
429
430        indices_a = (
431            np.broadcast_to(np.arange(n_active), (n_active, n_active))
432            + self._static_count
433        )
434
435        # Merge
436        full_dists = np.hstack((dists_s, dists_a_sq))
437        full_dists = np.maximum(full_dists, 0.0)
438        full_idxs = np.hstack((idxs_s, indices_a))
439
440        # Sort everything and take top k
441        k_final = min(k, full_dists.shape[1])
442        sort_order = np.argsort(full_dists, axis=1)[:, :k_final]
443
444        # Apply shuffle
445        final_dists_sq = np.take_along_axis(full_dists, sort_order, axis=1)
446        final_idxs = np.take_along_axis(full_idxs, sort_order, axis=1)
447
448        return final_idxs, np.sqrt(final_dists_sq)

450    def query_nn(self, k: int) -> tuple[np.ndarray, np.ndarray]:
451        """
452        Performs a hybrid k-NN search.
453
454        1. **Static Search:** Queries the Faiss index for $k$ neighbors.
455        2. **Active Search:** Computes exact distances against the active batch (brute-force).
456        3. **Merge:** Concatenates results and selects the global top-$k$.
457
458        Args:
459            k (int): Neighbors count.
460
461        Returns:
462            tuple[np.ndarray, np.ndarray]: Global indices and distances.
463        """
464        n_active = self._active.shape[0]
465
466        # 1. Query Static
467        k_s = min(k, self._static_count)
468        if k_s > 0:
469            dists_s, idxs_s = self._index_static.search(self._active, k_s)
470            dists_s = np.maximum(dists_s, 0.0)
471        else:
472            dists_s = np.empty((n_active, 0), dtype=np.float32)
473            idxs_s = np.empty((n_active, 0), dtype=int)
474
475        return self._merge_active_active_knn(dists_s, idxs_s, k)

477    def query_static(
478        self, query_points: np.ndarray, k: int
479    ) -> tuple[np.ndarray, np.ndarray]:
480        """
481        Directly queries the underlying Faiss index.
482
483        Args:
484            query_points (np.ndarray): Query inputs.
485            k (int): Neighbors count.
486
487        Returns:
488            tuple[np.ndarray, np.ndarray]: Indices and distances.
489        """
490        if self._static_count == 0:
491            return (
492                np.zeros((len(query_points), k), dtype=int),
493                np.full((len(query_points), k), np.inf),
494            )
495
496        q_data = np.ascontiguousarray(query_points.astype(np.float32))
497        dists, idxs = self._index_static.search(q_data, k)
498        return idxs, np.sqrt(np.maximum(dists, 0.0))

500    def _merge_active_active_range(
501        self, lims_s, D_s, I_s, radius: float
502    ) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
503        n_active = self._active.shape[0]
504        radius_sq = radius * radius
505
506        # Active-Active (Brute Force)
507        A_sq = np.sum(self._active**2, axis=1, keepdims=True)
508        dists_a_sq = A_sq + A_sq.T - 2 * np.dot(self._active, self._active.T)
509        dists_a_sq = np.maximum(dists_a_sq, 0.0)
510
511        mask_a = dists_a_sq < radius_sq
512
513        final_D_list = []
514        final_I_list = []
515
516        indices_a_base = np.arange(n_active) + self._static_count
517
518        for i in range(n_active):
519            s_start, s_end = lims_s[i], lims_s[i + 1]
520            d_s = D_s[s_start:s_end]
521            i_s = I_s[s_start:s_end]
522
523            a_mask = mask_a[i]
524            d_a = dists_a_sq[i][a_mask]
525            i_a = indices_a_base[a_mask]
526
527            d_combined = np.concatenate((d_s, d_a))
528            i_combined = np.concatenate((i_s, i_a))
529
530            final_D_list.append(np.sqrt(np.maximum(d_combined, 0.0)))
531            final_I_list.append(i_combined)
532
533        lens = [len(x) for x in final_D_list]
534        lims = np.zeros(n_active + 1, dtype=int)
535        lims[1:] = np.cumsum(lens)
536
537        if sum(lens) > 0:
538            D = np.concatenate(final_D_list)
539            Idx = np.concatenate(final_I_list)
540        else:
541            D = np.array([], dtype=np.float32)
542            Idx = np.array([], dtype=int)
543
544        return lims, D, Idx

546    def range_search(self, radius: float) -> tuple[np.ndarray, np.ndarray]:
547        """
548        Performs a hybrid range search returning a dense matrix.
549
550        This method is critical for performance in the physics simulation. It merges:
551        1. **Sparse Static Results:** `index.range_search()` returns a packed list of neighbors.
552            These are scattered into a dense matrix.
553        2. **Dense Active Results:** Brute-force calculation for the active batch is filled
554            into the remaining columns.
555
556        The result is a unified dense matrix $(M, N_{total})$ compatible with the vector
557        force equations.
558
559        Args:
560            radius (float): Cutoff radius.
561
562        Returns:
563            tuple[np.ndarray, np.ndarray]: (Distances Matrix, Validity Mask).
564        """
565        n_active = self._active.shape[0]
566        n_static = self._static_count
567        n_total = n_static + n_active
568
569        # 1. Allocate Unified Buffers (Batch x Total)
570        # Columns [0 : n_static] -> Static Neighbors
571        # Columns [n_static : n_total] -> Active Neighbors
572        dists_matrix = np.zeros((n_active, n_total), dtype=np.float32)
573        mask_matrix = np.zeros((n_active, n_total), dtype=bool)
574
575        # --- PART A: Static Neighbors (Faiss) ---
576        if n_static > 0:
577            lims, D, Idx = self._index_static.range_search(
578                self._active, radius * radius
579            )
580            if len(D) > 0:
581                counts = np.diff(lims.astype(np.int64))
582                row_indices = np.repeat(np.arange(n_active), counts)
583                col_indices = Idx
584                dists_matrix[row_indices, col_indices] = np.sqrt(D)
585                mask_matrix[row_indices, col_indices] = True
586
587        # --- PART B: Active-Active Neighbors (Numpy) ---
588        # Compute distances for the active batch against itself
589        # Shape: (Batch, Batch)
590        A_sq = np.sum(self._active**2, axis=1, keepdims=True)
591        dists_sq_active = A_sq + A_sq.T - 2 * np.dot(self._active, self._active.T)
592
593        # Avoid self-interaction and respect radius
594        # Note: We rely on the caller (physics engine) to handle epsilon/division-by-zero,
595        # but we must ensure self-interaction (dist=0) is masked out.
596        radius_sq = radius * radius

252        """

Performs a hybrid range search returning a dense matrix.

This method is critical for performance in the physics simulation. It merges:

  1. Sparse Static Results: index.range_search() returns a packed list of neighbors. These are scattered into a dense matrix.
  2. Dense Active Results: Brute-force calculation for the active batch is filled into the remaining columns.

The result is a unified dense matrix $(M, N_{total})$ compatible with the vector force equations.
